183  {
184  for (FrameId frame = 0; frame < numBufs; frame++)
185  { // for each frame in the buffer pool.
186  if (bufDescTable[frame].file == file)
187  { // if the frame belongs to the file.

189  { // if the frame is not valid, throw an exception.

191  }
192  if (bufDescTable[frame].pinCnt > 0)
193  { // if the frame is pinned, throw an exception.

195  }

199  bufDescTable[frame].dirty = false;
200  }
201  hashTable->remove(file, bufDescTable[frame].pageNo); // remove the frame from the hash table.
